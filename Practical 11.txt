    /* --- Check prerequisites before asking --- */
    function checkGeoPrerequisites() {
      // 1. API exists?
      if (!navigator.geolocation) {
        return { ok: false, reason: 'unsupported' };
      }
      // 2. Secure context?
      if (!window.isSecureContext) {
        return { ok: false, reason: 'insecure' };
      }
      // 3. Check permission state via Permissions API (if available)
      if (navigator.permissions) {
        return navigator.permissions.query({ name: 'geolocation' }).then(result => {
          if (result.state === 'denied') {
            return { ok: false, reason: 'denied' };
          }
          return { ok: true, reason: result.state };
        });
      }
      return { ok: true, reason: 'unknown' };
    }

    /* --- Radar Canvas Drawing --- */
    function drawRadar() {
      const w = geoCanvas.width / window.devicePixelRatio;
      const h = geoCanvas.height / window.devicePixelRatio;
      const cx = w / 2;
      const cy = h / 2;
      const maxR = Math.min(cx, cy) * 0.88;

      geoCtx.clearRect(0, 0, w, h);

      // Background
      geoCtx.fillStyle = '#080810';
      geoCtx.fillRect(0, 0, w, h);

      // Concentric rings
      for (let i = 1; i <= 4; i++) {
        const r = Math.max(1, (maxR / 4) * i);
        geoCtx.beginPath();
        geoCtx.arc(cx, cy, r, 0, Math.PI * 2);
        geoCtx.strokeStyle = `rgba(6, 182, 212, ${0.06 + i * 0.02})`;
        geoCtx.lineWidth = 1;
        geoCtx.stroke();
      }

      // Crosshairs
      geoCtx.strokeStyle = 'rgba(6, 182, 212, 0.08)';
      geoCtx.lineWidth = 1;
      geoCtx.beginPath();
      geoCtx.moveTo(cx - maxR, cy); geoCtx.lineTo(cx + maxR, cy);
      geoCtx.moveTo(cx, cy - maxR); geoCtx.lineTo(cx, cy + maxR);
      geoCtx.stroke();

      // Diagonals
      const diag = maxR * 0.707;
      geoCtx.beginPath();
      geoCtx.moveTo(cx - diag, cy - diag); geoCtx.lineTo(cx + diag, cy + diag);
      geoCtx.moveTo(cx + diag, cy - diag); geoCtx.lineTo(cx - diag, cy + diag);
      geoCtx.stroke();

      // Sweep trail
      radarAngle += 0.015;
      for (let a = 0; a < 0.5; a += 0.02) {
        const alpha = (0.5 - a) * 0.25;
        geoCtx.beginPath();
        geoCtx.moveTo(cx, cy);
        geoCtx.arc(cx, cy, maxR, radarAngle - a, radarAngle - a + 0.025);
        geoCtx.closePath();
        geoCtx.fillStyle = `rgba(6, 182, 212, ${alpha})`;
        geoCtx.fill();
      }

      // Sweep line
      geoCtx.beginPath();
      geoCtx.moveTo(cx, cy);
      geoCtx.lineTo(cx + Math.cos(radarAngle) * maxR, cy + Math.sin(radarAngle) * maxR);
      geoCtx.strokeStyle = 'rgba(6, 182, 212, 0.7)';
      geoCtx.lineWidth = 2;
      geoCtx.stroke();

      // Particles / blips
      geoParticles.forEach(p => {
        p.life -= 0.004;
        if (p.life <= 0) return;
        const px = cx + p.x * maxR;
        const py = cy + p.y * maxR;
        const size = Math.max(0.5, 3 * p.life);
        geoCtx.beginPath();
        geoCtx.arc(px, py, size, 0, Math.PI * 2);
        const col = isDemoMode ? `rgba(249, 115, 22, ${p.life * 0.8})` : `rgba(6, 182, 212, ${p.life * 0.8})`;
        geoCtx.fillStyle = col;
        geoCtx.fill();
      });
      geoParticles = geoParticles.filter(p => p.life > 0);

      // Location dot
      if (geoData) {
        const pulsePhase = (Date.now() % 2000) / 2000;
        const pulseR = Math.max(1, 8 + pulsePhase * 22);
        const dotColor = isDemoMode ? 'rgba(249, 115, 22,' : 'rgba(6, 182, 212,';
        const dotSolid = isDemoMode ? '#f97316' : '#06b6d4';

        // Outer pulse
        geoCtx.beginPath();
        geoCtx.arc(cx, cy, pulseR, 0, Math.PI * 2);
        geoCtx.strokeStyle = `${dotColor} ${0.6 * (1 - pulsePhase)})`;
        geoCtx.lineWidth = 2;
        geoCtx.stroke();

        // Second pulse ring
        const pulse2 = ((Date.now() + 1000) % 2000) / 2000;
        const pulse2R = Math.max(1, 8 + pulse2 * 22);
        geoCtx.beginPath();
        geoCtx.arc(cx, cy, pulse2R, 0, Math.PI * 2);
        geoCtx.strokeStyle = `${dotColor} ${0.3 * (1 - pulse2)})`;
        geoCtx.lineWidth = 1.5;
        geoCtx.stroke();

        // Center dot
        geoCtx.beginPath();
        geoCtx.arc(cx, cy, 6, 0, Math.PI * 2);
        geoCtx.fillStyle = dotSolid;
        geoCtx.fill();
        geoCtx.beginPath();
        geoCtx.arc(cx, cy, 2.5, 0, Math.PI * 2);
        geoCtx.fillStyle = '#fff';
        geoCtx.fill();

        // Coordinate text
        geoCtx.font = '500 11px "JetBrains Mono", monospace';
        geoCtx.fillStyle = isDemoMode ? 'rgba(249, 115, 22, 0.9)' : 'rgba(6, 182, 212, 0.9)';
        geoCtx.textAlign = 'center';
        geoCtx.fillText(`${geoData.lat.toFixed(5)}, ${geoData.lng.toFixed(5)}`, cx, cy + 28);
        geoCtx.fillStyle = 'rgba(255,255,255,0.3)';
        geoCtx.fillText(`ACC: ${geoData.acc.toFixed(0)}m`, cx, cy + 44);

        if (isDemoMode) {
          geoCtx.fillStyle = 'rgba(249, 115, 22, 0.5)';
          geoCtx.font = '600 9px "Space Grotesk", sans-serif';
          geoCtx.fillText('DEMO MODE', cx, cy - 18);
        }

        // Spawn blips
        if (Math.random() < 0.04) {
          const angle = Math.random() * Math.PI * 2;
          const dist = 0.15 + Math.random() * 0.7;
          geoParticles.push({ x: Math.cos(angle) * dist, y: Math.sin(angle) * dist, life: 1 });
        }
      }

      // Outer ring
      geoCtx.beginPath();
      geoCtx.arc(cx, cy, maxR, 0, Math.PI * 2);
      geoCtx.strokeStyle = isDemoMode ? 'rgba(249, 115, 22, 0.15)' : 'rgba(6, 182, 212, 0.15)';
      geoCtx.lineWidth = 2;
      geoCtx.stroke();

      // Tick marks
      for (let i = 0; i < 36; i++) {
        const a = (i / 36) * Math.PI * 2;
        const inner = i % 3 === 0 ? maxR - 10 : maxR - 5;
        geoCtx.beginPath();
        geoCtx.moveTo(cx + Math.cos(a) * inner, cy + Math.sin(a) * inner);
        geoCtx.lineTo(cx + Math.cos(a) * maxR, cy + Math.sin(a) * maxR);
        geoCtx.strokeStyle = isDemoMode ? 'rgba(249, 115, 22, 0.2)' : 'rgba(6, 182, 212, 0.2)';
        geoCtx.lineWidth = i % 3 === 0 ? 2 : 1;
        geoCtx.stroke();
      }

      geoAnimFrame = requestAnimationFrame(drawRadar);
    }

    /* ============================================
       WEB STORAGE API
    ============================================ */
    const STORAGE_KEYS = {
      local: 'html5_demo_local_notes',
      session: 'html5_demo_session_notes'
    };

    function getNotes(type) {
      const storage = type === 'local' ? localStorage : sessionStorage;
      try { return JSON.parse(storage.getItem(STORAGE_KEYS[type])) || []; }
      catch { return []; }
    }

    function setNotes(type, notes) {
      const storage = type === 'local' ? localStorage : sessionStorage;
      storage.setItem(STORAGE_KEYS[type], JSON.stringify(notes));
      renderNotes(type);
      updateInspector();
    }

    function renderNotes(type) {
      const list = document.getElementById(`${type}-notes-list`);
      const notes = getNotes(type);
      if (notes.length === 0) {
        list.innerHTML = '<div class="empty-state"><i class="fas fa-inbox"></i>No notes saved yet</div>';
        return;
      }
      list.innerHTML = notes.map((note, i) => `
        <div class="note-item">
          <div style="flex:1">
            <div class="note-text">${escapeHtml(note.text)}</div>
            <div class="note-time">${note.time}</div>
          </div>
          <button class="note-delete" data-type="${type}" data-index="${i}" aria-label="Delete note" title="Delete">
            <i class="fas fa-xmark"></i>
          </button>
        </div>
      `).join('');
    }

    function escapeHtml(str) {
      const div = document.createElement('div');
      div.textContent = str;
      return div.innerHTML;
    }

    function addNote(type) {
      const input = document.getElementById(`${type}-input`);
      const text = input.value.trim();
      if (!text) { showToast('Please type a note first', 'warning'); input.focus(); return; }
      const notes = getNotes(type);
      notes.unshift({ text, time: new Date().toLocaleString() });
      setNotes(type, notes);
      input.value = '';
      input.focus();
      showToast(`Note saved to ${type === 'local' ? 'localStorage' : 'sessionStorage'}`, 'success');
    }

    document.addEventListener('click', (e) => {
      const btn = e.target.closest('.note-delete');
      if (!btn) return;
      const notes = getNotes(btn.dataset.type);
      notes.splice(parseInt(btn.dataset.index), 1);
      setNotes(btn.dataset.type, notes);
      showToast('Note deleted', 'warning');
    });

    document.getElementById('local-save-btn').addEventListener('click', () => addNote('local'));
    document.getElementById('session-save-btn').addEventListener('click', () => addNote('session'));
    document.getElementById('local-input').addEventListener('keydown', (e) => { if (e.key === 'Enter') addNote('local'); });
    document.getElementById('session-input').addEventListener('keydown', (e) => { if (e.key === 'Enter') addNote('session'); });

    document.getElementById('local-clear-btn').addEventListener('click', () => {
      localStorage.removeItem(STORAGE_KEYS.local);
      renderNotes('local'); updateInspector();
      showToast('localStorage cleared', 'warning');
    });
    document.getElementById('session-clear-btn').addEventListener('click', () => {
      sessionStorage.removeItem(STORAGE_KEYS.session);
      renderNotes('session'); updateInspector();
      showToast('sessionStorage cleared', 'warning');
    });

    document.getElementById('local-export-btn').addEventListener('click', () => exportNotes('local'));
    document.getElementById('session-export-btn').addEventListener('click', () => exportNotes('session'));

    function exportNotes(type) {
      const notes = getNotes(type);
      if (notes.length === 0) { showToast('Nothing to export', 'warning'); return; }
      const blob = new Blob([JSON.stringify(notes, null, 2)], { type: 'application/json' });
      const url = URL.createObjectURL(blob);
      const a = document.createElement('a');
      a.href = url; a.download = `${type}-notes-${Date.now()}.json`; a.click();
      URL.revokeObjectURL(url);
      showToast('Notes exported as JSON', 'success');
    }

    function updateInspector() {
      const container = document.getElementById('inspector-entries');
      const entries = [];
      for (let i = 0; i < localStorage.length; i++) {
        const key = localStorage.key(i);
        entries.push({ key, value: localStorage.getItem(key), type: 'local' });
      }
      for (let i = 0; i < sessionStorage.length; i++) {
        const key = sessionStorage.key(i);
        entries.push({ key, value: sessionStorage.getItem(key), type: 'session' });
      }
      if (entries.length === 0) {
        container.innerHTML = '<div class="empty-state" style="padding:0.5rem"><i class="fas fa-ghost"></i>No storage entries found</div>';
        return;
      }
      container.innerHTML = entries.map(e => `
        <div class="inspector-row">
          <span class="inspector-key">${escapeHtml(e.key)}</span>
          <span class="inspector-val">${escapeHtml(e.value.substring(0, 60))}${e.value.length > 60 ? '...' : ''}</span>
        </div>
      `).join('');
    }

    window.addEventListener('storage', () => { renderNotes('local'); updateInspector(); });

    // Initial render
    renderNotes('local');
    renderNotes('session');
    updateInspector();
  </script>
